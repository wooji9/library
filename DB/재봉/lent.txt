create sequence lent_idx;

create table lent(
	idx number(5) primary key,--일련번호
	m_id varchar2(50) not null,--회원ID
	b_isbl varchar2(50) not null,--책ID
	lent varchar2(50) not null,--예약대여여부
	lentdate date not null, --대여일
	redate date,
	b_name varchar2(50) not null 
);
